# 请求封装

# Axios的二次封装

​ 为了统一管理请求的处理，比如要进行一些统一配置请求信息，统一进行状态码处理，loading状态处理，都需要将axios进行二次封装，方便日后维护和管理啊。

import axios from 'axios'
import { getToken } from './auth'

// 错误统一处理
const errorHandler = (error) => {
  if (error.response) {
    switch (error.response.status) {
      case 401:
        // 401 Unauthorized 缺少身份校验
        break;
      case 403:
        // 用户无权限, 服务器拒绝访问
        break;
      case 404:
        // Not Found
        break;
      case 500:
        // 服务器错误
        break;
      case 504:
        // 网络超时
        break;
      default:
        // 其它错误码，提示信息
        break
    }
  }
  return Promise.reject(error)
}


// axios实例
const instance = axios.create({
  baseURL: process.env.VUE_APP_BASE_API, // 基础路由
  timeout: 5000, // timeout
})

// 请求拦截器
instance.interceptors.request.use(
  config => {
    if (store.getters.token) {
      // 请求携带token请求头信息
      // ['X-Token'] is a custom headers key
      config.headers['X-Token'] = getToken()
    }
    return config
  },
  // 错误处理函数
  errorHandler
)

instance.interceptors.response.use(
  response => {
    // todo: 请求成功的其它统一处理，比如成功但返回code: 400这种处理
    return response.data
  },
  // 错误处理函数
  errorHandler
)

export default instance

# API封装

​ 一般来说，页面会随着大量的请求，导致接口和页面形成耦合，每次修改了接口名称，都需要找到对应的页面进行接口名称的修改，有时候避免不了遗漏的错误，所以我们可以抽离API接口名称，单独管理，需要请求时，引入对应的方法进行请求。

例子：用户模块接口

@/api/user.js

import request from '@/utils/request'

export function login(data) {
  return request({
    url: '/user/login',
    method: 'post',
    data
  })
}

export function getInfo(token) {
  return request({
    url: '/user/info',
    method: 'get',
    params: { token }
  })
}

export function logout() {
  return request({
    url: '/user/logout',
    method: 'post'
  })
}
